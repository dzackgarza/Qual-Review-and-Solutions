\subsection{Exercises}

\begin{problem}[Hungerford 7.3.1]
\label{prob:1.1}
Let $B$ be an $R$-module. Show that if $r+r\neq 0$ for all $r\neq 0 \in R$, then an $n$-linear form $B^n\to R$ is alternating $\iff$ it is skew-symmetric.
\end{problem}

\begin{problem}[Hungerford 7.3.5]
\label{prob:1.1}
If $R$ is a field and $A, B \in M_n(R)$ are invertible then the matrix $A + rB$ is invertible for all but a finite number of $r\in R$.
\end{problem}

\begin{problem}[Hungerford 7.4.4]
\label{prob:1.1}
Show that if $q$ is the minimal polynomial of a linear transformation $\phi: E\to E$ with $\dim_k E = n$ then $\deg q \leq n$.
\end{problem}

\begin{problem}[Hungerford 7.4.8]
\label{prob:1.1}
Show that $A\in M_n(K)$ is similar to a diagonal matrix $\iff$ the elementary divisors of $A$ are all linear.
\end{problem}

\begin{problem}[Hungerford 7.4.10]
\label{prob:1.1}
Find all possible rational canonical forms for a matrix $A\in M_n(\Bbb Q)$ such that 

\begin{enumerate}
    \item $A$ is $6\times 6$ with minimal polynomial $q(x) = (x-2)^2(x+3)$.
    \item $A$ is $7\times 7$ with $q(x) = (x^2+1)(x-7)$.
\end{enumerate}

Also find all such forms when $A \in M_n(\Bbb C)$ instead, and find all possible Jordan Canonical Forms over $\Bbb C$.
\end{problem}

\begin{problem}[Hungerford 7.5.2]
\label{prob:1.1}
Show that if $\phi$ is an endomorphism of a free $k$-module $E$ of finite rank, then $p_\phi(\phi) = 0$.

\textit{
Hint: \\
If $A$ is the matrix of $\phi$ and $B = x I_n - A$ then $B^a B = \abs{B} I_n = p_\phi I_n$ in $M_n(k[x])$. \\
If $E$ is a $k[x]$-module with structure induced by $\phi$, and $\psi$ is the $k[x]$-module endomorphism $E\to E$ with matrix given by $B$, then $$\psi(u) = x u -\phi(u) = \phi(u) - \phi(u) = 0$$ for all $u\in E$.
}
\end{problem}

\begin{problem}[Hungerford 7.5.7]
\label{prob:1.1}
\hfill
\begin{enumerate}
    \item Let $\phi,\psi$ be endomorphisms of a finite-dimensional vector space $E$ such that $\phi\psi = \psi \phi$. Show that if $E$ has a basis of eigenvectors of $\psi$, then it has a basis of eigenvectors for both $\psi$ and $\phi$ simultaneously.
    
    \item Interpret the previous part as a statement about matrices similar to a diagonal matrix.
\end{enumerate}
\end{problem}

\subsection{Qual Problems}

\begin{problem}
\label{prob:1.1}
Let $M \in M_5(R)$ be a $5\times 5$ square matrix with real coefficients defining a linear map $L: \Bbb{R}^5 \to \Bbb R^5$.
Assume that when considered as an element of $M_5(\Bbb C)$, then the scalars $0, 1+i, 1+2i$ are eigenvalues of $M$.
\begin{enumerate}
    \item Show that the associated linear map $L$ is neither injective nor surjective.
    \item Compute the characteristic polynomial and minimal polynomial of $M$.
    \item How many fixed points can $L$ have?
    
    \textit{(That is, how many solutions are there to the equation $L(v) = v$ with $v\in \Bbb R^5$?)}
\end{enumerate}
\end{problem}

\begin{problem}
\label{prob:1.1}
Let $n$ be a positive integer and let $B$ denote the $n\times n$ matrix over $\Bbb C$ such that every entry is 1. Find the Jordan normal form of $B$.
\end{problem}

\begin{problem}
\label{prob:1.1}
Suppose that $V$ is a 6-dimensional vector space and that $T$ is a linear transformation on $V$ such that $T^6 = 0$ and $T^5\neq 0$.

\begin{enumerate}
    \item Find a matrix for $T$ in Jordan Canonical form.
    \item Show that if $S, T$ are linear transformations on a 6-dimensional vector space $V$ which both satisfy $T^6=S^6=0$ and $T^5,S^5\neq 0$, then there exists a linear transformation $A$ from $V$ to itself such that $ATA^{-1} = S$.
\end{enumerate}
\end{problem}
